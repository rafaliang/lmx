for $tuple in join(
	for $a in doc("./src/a.xml")//actor,
		$aa in $a/age
		return <tuple>{<a>{$a}</a>,<aa>{$aa}</aa>}</tuple>,
	for $s in doc("./src/a.xml")//singer,
		$sa in $s/age
		return <tuple>{<s>{$s}</s>,<sa>{$sa}</sa>}</tuple>,
	[aa],[sa])
return
<people-with-ages>{
	$tuple/aa/*,
	<actor>{$tuple/a/*}</actor>,
	<singer>{$tuple/s/*}</singer>}
</people-with-ages>
